import { NextResponse } from "next/server";
import OpenAI from "openai";
import { prisma } from "@/lib/prisma";
import { auth } from "@/auth";
import { buildQuickPrompt } from "./prompts";

type Mode = "quick" | "report";
type Access = "FREE" | "PAY_PER_REPORT" | "SUBSCRIBER";

const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

function numEnv(name: string, fallback: number) {
  const v = Number(process.env[name] ?? "");
  return Number.isFinite(v) && v > 0 ? v : fallback;
}

function monthKeyUTC(d = new Date()) {
  return d.toISOString().slice(0, 7); // YYYY-MM
}

function modelFor(access: Access, economy: boolean) {
  if (economy) return process.env.MECHVI_ECONOMY_MODEL || "gpt-5-nano";
  if (access === "FREE") return process.env.MECHVI_MODEL_FREE || "gpt-5-nano";
  if (access === "PAY_PER_REPORT") return process.env.MECHVI_MODEL_REPORT || "gpt-5-mini";
  return process.env.MECHVI_MODEL_SUBSCRIBER || "gpt-5.2";
}

function maxOutFor(access: Access, economy: boolean, mode: Mode) {
  if (economy) return numEnv("MECHVI_ECONOMY_MAX_OUT", 450);
  if (access === "FREE") return mode === "quick" ? 650 : 900;
  if (access === "PAY_PER_REPORT") return 1900;
  return 2200;
}

function paywallResponse(mode: Mode) {

  return {

// Robustly extract JSON from model text
function extractJson(text: string): any {
  const t = text.trim();
  if (!t) throw new Error("Empty model output");

  // If it's already JSON
  if (t.startsWith("{") && t.endsWith("}")) return JSON.parse(t);

  // Try to find first {...} block
  const first = t.indexOf("{");
  const last = t.lastIndexOf("}");
  if (first >= 0 && last > first) {
    const slice = t.slice(first, last + 1);
    return JSON.parse(slice);
  }

  throw new Error("Could not parse JSON from model output");
}

function ensureQuickCards(parsed: any) {
  const cards = Array.isArray(parsed.cards) ? parsed.cards : [];

  const mkCard = (type: string, title: string) => ({
    type,
    title,
    severity: "info",
    bullets: [],
    table: { columns: [], rows: [] },
    score: { label: "", value: null, scale_max: 3, bands: [] },
    actions: [],
  });

  const get = (type: string, title: string) => {
    let c = cards.find((x: any) => x?.type === type);
    if (!c) {
      c = mkCard(type, title);
      cards.push(c);
    }
    c.title = title;
    c.severity = c.severity || "info";
    c.bullets = Array.isArray(c.bullets) ? c.bullets : [];
    c.actions = [];
    return c;
  };

  const summary = get("summary", "Quick Verdict");

  // Normalize fit
  const raw = Number(summary?.score?.value);
  const norm = raw === 3 ? 3 : raw === 1 ? 1 : 2;
  const fitName = norm === 3 ? "Good ✅" : norm === 2 ? "Mixed 🤔" : "Lemon 🍋";

  const cleaned = (summary.bullets as any[])
    .map((b) => String(b || "").trim())
    .filter(Boolean)
    .filter((b) => !/^fit:\s*/i.test(b));

  const extra = cleaned.slice(0, 2);
  while (extra.length < 2) extra.push("Mechanic-style guidance, not guesswork.");

  summary.bullets = [`Fit: ${fitName}`, extra[0], extra[1]];
  summary.score = summary.score || {};
  summary.score.label = "Fit";
  summary.score.value = norm;
  summary.score.scale_max = 3;
  summary.score.bands = [
    { name: "Good", min: 3, max: 3 },
    { name: "Mixed", min: 2, max: 2 },
    { name: "Lemon", min: 1, max: 1 },
  ];

  const risk = get("risk_flags", "Common Problems / Red Flags");
  if (!risk.bullets.length) {
    risk.bullets = [
      "Service history is king — missed servicing = hidden problems.",
      "Check for leaks, overheating signs, smoke, and weird driveline noises.",
      "If diesel: watch for DPF/EGR issues (short trips) + rough idle.",
    ];
  }
  risk.bullets = (risk.bullets as any[]).slice(0, 3);

  const costs = get("costs", "Fuel & Running Costs");
  if (!costs.bullets.length) {
    costs.bullets = [
      "Real fuel use depends on city vs highway and load/towing.",
      "Tyres + brakes are the big predictable costs.",
      "Neglect + short trips (diesel) = expensive surprises.",
    ];
  }
  costs.bullets = (costs.bullets as any[]).slice(0, 3);

  const checklist = get("checklist", "Before You Buy (5-minute check)");
  if ((checklist.bullets as any[]).length < 5) {
    checklist.bullets = [
      "Tyres + brakes: look for uneven wear + thin pads through the wheel.",
      "Open bonnet: check oil level/colour + coolant level (no sludge).",
      "Cold start: listen for rattles/knocks; watch for smoke.",
      "Test drive: straight braking, smooth shifts, no vibrations.",
      "Paperwork: service history + PPSR/rego + scan for fault codes.",
    ];
  }
  checklist.bullets = (checklist.bullets as any[]).slice(0, 5);

  // Remove generic next_steps if present
  parsed.cards = cards.filter((c: any) => c?.type !== "next_steps");
  return parsed;
}

export async function POST(req: Request) {
  try {
    const session = await auth();
    const email = session?.user?.email ? String(session.user.email).toLowerCase() : null;

    const body = await req.json();
    const mode: Mode = (body.mode || "quick") as Mode;

    const vp = body.vehicleProfile || {};
    const make = String(vp.make || "").trim();
    const model = String(vp.model || "").trim();
    const year = Number(vp.year || 0);
    const transmission = String(vp.transmission || "unknown");
    const fuel = String(vp.fuel || "unknown");
    const cylinders = String(vp.cylinders || "unknown");
    const km = vp.km === null || vp.km === undefined ? null : Number(vp.km);
    const locale = String(body.locale || "en-AU");

    if (!make || !model || !year) {
      return NextResponse.json({ error: "Missing make/model/year" }, { status: 400 });
    }

    // Determine access
    let access: Access = "FREE";
    let userId: string | null = null;

    if (email) {
      const u = await prisma.user.findUnique({ where: { email } });
      if (u) {
        userId = u.id;
        access = u.tier === "SUBSCRIBER" ? "SUBSCRIBER" : "FREE";
      }
    }

    // Usage limits
    const mk = monthKeyUTC();
    const freeLimit = numEnv("MECHVI_FREE_MONTHLY_LIMIT", 3);
    const subLimit = numEnv("MECHVI_SUBSCRIBER_MONTHLY_LIMIT", 300);

    let usageCount = 0;
    let economy = false;

    if (userId) {
      const row = await prisma.usage.upsert({
        where: { userId_monthKey: { userId, monthKey: mk } },
        update: {},
        create: { userId, monthKey: mk, count: 0 },
      });
      usageCount = row.count;

      if (access === "FREE" && mode === "quick" && usageCount >= freeLimit) {
        return NextResponse.json(paywallResponse("quick"), { status: 200 });
      }

      if (access === "SUBSCRIBER" && usageCount >= subLimit) {
        economy = true;
      }
    } else {
      // No login → treat as FREE but allow (for now)
      if (mode === "report") {
        // report is paid feature in production; keep placeholder paywall for anonymous
        return NextResponse.json(paywallResponse("report"), { status: 200 });
      }
    }

    // Increment usage for logged-in users (only when we actually call the model)
    if (userId) {
      await prisma.usage.update({
        where: { userId_monthKey: { userId, monthKey: mk } },
        data: { count: { increment: 1 } },
      });
    }

    const vehicleLabel = `${year} ${make} ${model}`.trim();

    const prompt =
      mode === "quick"
        ? buildQuickPrompt({ make, model, year, transmission, fuel, cylinders, km, locale })
        : `
You are MechVi — your mechanic in your pocket (Australia).
Write for a total beginner. Practical, no fluff.

Vehicle:
- ${vehicleLabel}
- Transmission=${transmission}, Fuel=${fuel}, Cylinders=${cylinders}, Km=${km ?? "unknown"}

Return JSON ONLY in the same schema as quick mode, but more detailed:
Include cards: summary, risk_flags, costs, checklist, plan.
In checklist, be step-by-step and detailed for a beginner.
In plan, give first 30 days actions + what to service first.
`.trim();

    const modelName = modelFor(access, economy);
    const max_output_tokens = maxOutFor(access, economy, mode);

    const resp = await client.responses.create({
      model: modelName,
      input: prompt,
      max_output_tokens,
    });

    const outText = (resp as any).output_text || "";
    let parsed = extractJson(outText);

    // Enforce meta and entitlements shape
    parsed.meta = parsed.meta || {};
    parsed.meta.assistant = "mechvi";
    parsed.meta.version = "1.2";
    parsed.meta.access_level = access;
    parsed.meta.locale = locale;
    parsed.meta.intent = "buy";
    parsed.meta.disclaimer_level = "light";

    parsed.entitlements = parsed.entitlements || {};
    parsed.entitlements.garage = { can_save_vehicle: true, free_slots_total: 1, free_slots_used: 0 };
    parsed.entitlements.alerts = { enabled: access === "SUBSCRIBER" };
    parsed.entitlements.logbook = { enabled: access === "SUBSCRIBER" };
    parsed.entitlements.personalised_plan = { enabled: access === "SUBSCRIBER" };

    // CTA rules
    if (mode === "report" && access !== "SUBSCRIBER") {
      parsed.cta = { label: "Get $2.99 Report (placeholder)", action_type: "paywall", payload: { product: "mechvi_report", price_aud: 2.99 } };
    } else if (access !== "SUBSCRIBER") {
      parsed.cta = { label: "Subscribe $7.99/m (placeholder)", action_type: "subscribe", payload: { plan: "plus", price_aud_per_month: 7.99 } };
    } else {
      parsed.cta = { label: "Save to Garage (placeholder)", action_type: "save_vehicle", payload: { vehicle: { make, model, year, transmission, fuel, cylinders, km } } };
    }

    // Enforce better FREE quick structure
    if (mode === "quick") {
      parsed = ensureQuickCards(parsed);
    }

    return NextResponse.json(parsed, { status: 200 });
  } catch (e: any) {
    return NextResponse.json(
      { error: e?.message || "Ask failed" },
      { status: 500 }
    );
  }
}

export function buildQuickPrompt(input: {
  make: string;
  model: string;
  year: number;
  transmission: string;
  fuel: string;
  cylinders: string;
  km: number | null;
  locale: string;
}) {
  const { make, model, year, transmission, fuel, cylinders, km, locale } = input;

  return `
You are MechVi — "your mechanic in your pocket" for Australia.
You MUST write like a real senior mechanic talking to a normal person (friendly, confident, practical).
No corporate fluff. No “research financing”. No dealership talk.
Use short, punchy sentences. Helpful, slightly cheeky is OK.

Vehicle:
- Make: ${make}
- Model: ${model}
- Year: ${year}
- Transmission: ${transmission}
- Fuel: ${fuel}
- Cylinders: ${cylinders}
- Km: ${km ?? "unknown"}

Task:
Return JSON ONLY using the existing response schema. Populate exactly these 4 cards, in order:

1) type="summary", title="Quick Verdict"
   - bullets: exactly 3 bullets:
     a) First bullet MUST be "Fit: Good ✅" or "Fit: Mixed 🤔" or "Fit: Lemon 🍋"
     b) Second bullet: One-liner verdict vibe (e.g. "Workhorse ute. Built to cop abuse." / "Good if maintained, but watch the weak spots.")
     c) Third bullet: Who it suits (daily tradie? towing? family? city? off-road?)

   - score.value MUST be 1,2,or 3 corresponding to Lemon/Mixed/Good.

2) type="risk_flags", title="Common Problems / Red Flags"
   - bullets: 3 bullets max. Realistic red flags for THIS kind of car/year OR, if unknown/new model year, give universal checks (DPF, injector, cooling, turbo hoses, servicing, accident history).
   - Keep each bullet actionable (what to look for).

3) type="costs", title="Fuel & Running Costs"
   - bullets: 3 bullets max:
     - fuel use vibe for AU (city vs highway)
     - typical expensive consumables (tyres, brakes, services)
     - what makes costs spike (towing, short trips, neglect)

4) type="checklist", title="Before You Buy (5-minute check)"
   - bullets: exactly 5 bullets, step-by-step, for a total beginner.
     Include: tyres/brakes visual, oil/coolant check, test drive feel, smoke/overheat, service history & scan.

Important:
- If year is very new and data is uncertain, say so briefly but STILL give useful checks.
- No legal/medical stuff. No brand bashing. No calling the car junk without explanation.
- Locale: ${locale}. Use AU wording (km, ute, rego, etc).

Return JSON only.
`.trim();
}
